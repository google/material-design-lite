export const TYPES = [
  "display-large",
  "display-medium",
  "display-small",
  "headline-large",
  "headline-medium",
  "headline-small",
  "title-large",
  "title-medium",
  "title-small",
  "body-large",
  "body-medium",
  "body-small",
  "label-large",
  "label-medium",
  "label-small",
];

{
TYPES.map(type => {
const titleCase = type.split("-").map(word => word.charAt(0).toUpperCase() + word.slice(1)).join(" ");
return <div class={`pad-v ${type}`}>{titleCase}</div>;
})
}
